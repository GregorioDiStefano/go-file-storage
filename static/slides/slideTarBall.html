# Upload entire directories


$ tar -zcf - foo/ | curl --upload-file - "https://grabfile.io/foo.tar.gz"



# Upload filtered log files


$ grep 'error' /var/log/syslog | curl --upload-file - https://grabfile.io/logs.txt



# Encrypt and upload


$ gpg -o - -c --symmetric bar | curl --upload-file - https://grabfile.io/bar
